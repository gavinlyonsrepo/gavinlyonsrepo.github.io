179template <typename T>
180std::string formatElement(const T &element, int format) {
181 std::ostringstream oss;
182
183 if constexpr (std::is_integral_v<T>) {
184 // Handle integral types with base conversion
185 switch (format) {
186 case RDL_DEC:
187 oss << element; // Decimal
188 break;
189 case RDL_HEX:
190 oss << std::hex << std::uppercase << element; // Hexadecimal
191 break;
192 case RDL_OCT:
193 oss << std::oct << element; // Octal
194 break;
195 case RDL_BIN: {
196 // Binary conversion using std::bitset without leading zeros
197 std::string binary = std::bitset<sizeof(T) * 8>(element).to_string();
198 binary.erase(0, binary.find_first_not_of('0')); // Strip leading zeros
199 oss << (binary.empty() ? "0" : binary); // Handle the case for 0
200 break;
201 }
202 default:
203 oss << element; // Default to decimal
204 }
205 } else if constexpr (std::is_floating_point_v<T>) {
206 // Handle floating-point types with fixed precision
207 oss << std::fixed << std::setprecision(format) << element;
208 } else {
209 // Handle other types (fallback)
210 oss << element;
211 }
212
213 return oss.str();
214}
