146template <typename T>
147std::string formatElement(const T &element, int format) {
148 std::ostringstream oss;
149
150 if constexpr (std::is_integral_v<T>) {
151 // Handle integral types with base conversion
152 switch (format) {
153 case RDL_DEC:
154 oss << element; // Decimal
155 break;
156 case RDL_HEX:
157 oss << std::hex << std::uppercase << element; // Hexadecimal
158 break;
159 case RDL_OCT:
160 oss << std::oct << element; // Octal
161 break;
162 case RDL_BIN: {
163 // Binary conversion using std::bitset without leading zeros
164 std::string binary = std::bitset<sizeof(T) * 8>(element).to_string();
165 binary.erase(0, binary.find_first_not_of('0')); // Strip leading zeros
166 oss << (binary.empty() ? "0" : binary); // Handle the case for 0
167 break;
168 }
169 default:
170 oss << element; // Default to decimal
171 }
172 } else if constexpr (std::is_floating_point_v<T>) {
173 // Handle floating-point types with fixed precision
174 oss << std::fixed << std::setprecision(format) << element;
175 } else {
176 // Handle other types (fallback)
177 oss << element;
178 }
179
180 return oss.str();
181}
